#include <stdio.h>
 int main(){
     int arr[] = {12, 8, 45,30, 98, 2,7,11};
     for(int i = sizeof(arr) / sizeof(arr[0]); i > 0 ; i--){
         for(int j = 0; j < i; j++){
            if(arr[j] > arr[j+1]){
                int tmp = arr[j];
                 arr[j] = arr[j+1];
                 arr[j+1] = tmp;
            }
         }
     }
     for(int k = 0; k < sizeof(arr) / sizeof(arr[0]); k++){
         printf("%d\n", arr[k]);
     }
 }
